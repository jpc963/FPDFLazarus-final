AcceptPageBreak

AcceptPageBreak: boolean;

Descrição

Sempre que uma condição de quebra de página é encontrada, o método é chamado, e a quebra ocorre ou não dependendo do valor retornado. A implementação padrão retorna um valor de acordo com o modo selecionado por SetAutoPageBreak().
Este método é chamado automaticamente e não deve ser chamado diretamente pela aplicação.

Exemplo

O método é sobreposto em uma classe herdada a fim obter um layout de 3 colunas: 

  Type
    TMeuPDF = class(TFPDF)
      private
        Coluna: integer;
        Procedure SetCol(AColuna: integer);
        function AcceptPageBreak: Boolean; override;
    end;
  
  { TMeuPDF }
  
  procedure TMeuPDF.SetCol(AColuna: integer);
  Var x: Integer;
  begin
    // Move a posição para a coluna especificada
    Self.Coluna := AColuna;
    x := 10+(AColuna*65);
    Self.SetLeftMargin(x);
    Self.SetX(x);
  end;
  
  function TMeuPDF.AcceptPageBreak: Boolean;
  begin
    if (Self.Coluna<2) then
    begin
      // Vai para a próxima coluna
      Self.SetCol(Self.Coluna+1);
      Self.SetY(10);
      Result := false;
    end
    else
    begin
      // Volta para a primeira coluna e permite a quebra de página
      Self.SetCol(0);
      Result := true;
    end;
  end;
  
  Var
    PDF : TMeuPDF;
    i: integer;
  begin
    try
  
      PDF := TMeuPDF.Create();
      PDF.AddPage;
      PDF.SetFont('Arial','',12);
      for i := 1 to 300 do
        PDF.Cell(0,5,'Linha '+IntToStr(i),'0',1);
      PDF.SaveToFile('teste.pdf');
  
    except
      on E: Exception do
        Writeln(E.ClassName, ': ', E.Message);
    end;
  end.
